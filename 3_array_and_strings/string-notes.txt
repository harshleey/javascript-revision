.toLowerCase() - Whisper mode
javascript

// Result: "hello world"

javascript

3. .includes() & .indexOf() - The Search Squad
.includes() - Simple yes/no search
javascript
const email = "user@example.com";
const hasAtSymbol = email.includes("@");
// Result: true (just tells you if it exists)

const sentence = "The quick brown fox";
const hasDog = sentence.includes("dog");
// Result: false
.indexOf() - Detective search (finds position)
javascript
const sentence = "I love JavaScript";
const jsPosition = sentence.indexOf("JavaScript");
// Result: 7 (position where it starts)

const notFound = sentence.indexOf("Python");
// Result: -1 (means not found)
Quick comparison:

javascript
const text = "hello world";

// Just need to know if it exists?
text.includes("world");  // true - simpler!

// Need the position?
text.indexOf("world");   // 6 - gives location!
4. .replace() & .substring() - The Text Surgeons
.replace() - Find and replace
javascript
// Replace first occurrence
const message = "I like cats. Cats are cute.";
const dogMessage = message.replace("cats", "dogs");
// Result: "I like dogs. Cats are cute."

// Replace ALL occurrences (use regex)
const allDogs = message.replace(/cats/gi, "dogs");
// Result: "I like dogs. Dogs are cute."
.substring() - Extract a portion
javascript
const fullText = "HelloWorld";

// Get from index 0 to 5 (not including 5)
const hello = fullText.substring(0, 5);
// Result: "Hello"

// Get from index 5 to end
const world = fullText.substring(5);
// Result: "World"

5. .trim() - The Space Cleaner
javascript
const userInput = "   hello world   ";
const clean = userInput.trim();
// Result: "hello world" (removes spaces from both ends)

// Also: .trimStart(), .trimEnd()
const leftClean = "   hello".trimStart(); // "hello"
const rightClean = "hello   ".trimEnd();  // "hello"
6. .startsWith() & .endsWith() - The Boundary Checkers
javascript
const filename = "document.pdf";

// Check beginning
const isDocument = filename.startsWith("document");
// Result: true

// Check end
const isPDF = filename.endsWith(".pdf");
// Result: true

const url = "https://example.com";
const isSecure = url.startsWith("https");
// Result: true
7. .slice() - The Substring Alternative
javascript
const text = "JavaScript";

// Similar to substring
const java = text.slice(0, 4);    // "Java"
const script = text.slice(4);     // "Script"

// But can use negative indexes (from end)
const last3 = text.slice(-3);     // "ipt"
8. .repeat() - The Duplicator
javascript
const laugh = "ha";

// Result: "hahaha"

const indent = " ".repeat(4); // 4 spaces
